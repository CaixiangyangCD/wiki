    接口¶

    定义¶

    Golang中的接口是一种新的类型定义，它把所有的 具有共性的方法 定义在一起，任何其他类型只要实现了这些方法就是实现了这个接口。

    接口的语法格式：

    // 定义接口
type interface_name interface {
    method_name1 [return_type]
    method_name2 [return_type]
    method_name3 [return_type]
    ...
    method_namen [return_type]
}

// 定义结构体
type struct_name struct {

}

// 实现接口方法
func (struct_name_variable struct_name) method_name1() [return_type] {
    // 方法实现
}
...
func (struct_name_variable struct_name) method_namen() [return_type] {
    // 方法实现
}

    实例¶

    定义一个USB接口，有读read和写write两个方法，再定义一个电脑Computer和一个手机Mobile来实现这个接口

    USB接口

    type USB interface {
    // 声明两个没有实现的方法read和write
    read()
    write()
}

    Computer结构体

    type Computer struct {
    name string
}

    Computer实现USB接口方法

    func (c Computer) read() {
    fmt.Printf("Computer %v is reading data\n", c.name)
    fmt.Println("reading...")
}

func (c Computer) write() {
    fmt.Printf("Computer %v is writing data\n", c.name)
    fmt.Println("writing...")
}

    Mobile结构体

    type Mobile struct {
    name string
}

    Mobile实现USB接口方法

    func (m Mobile) read() {
    fmt.Printf("Mobile %v is reading data\n", m.name)
    fmt.Println("reading...")
}

func (m Mobile) write() {
    fmt.Printf("Mobile %v is writing data\n", m.name)
    fmt.Println("writing...")
}

    完整实例：

    package main

import "fmt"

// 定义一个USB接口
type USB interface {
    // 声明两个没有实现的方法read和write
    read()
    write()
}

// 定义Computer结构体
type Computer struct {
    name string
}

// 使用Computer结构体实现USB接口的read方法
func (c Computer) read() {
    fmt.Printf("Computer %v is reading data\n", c.name)
    fmt.Println("reading...")
}

// 使用Computer结构体实现USB接口的write方法
func (c Computer) write() {
    fmt.Printf("Computer %v is writing data\n", c.name)
    fmt.Println("writing...")
}

// 定义一个Mobile结构体
type Mobile struct {
    name string
}

// 使用Mobile结构体实现USB接口的read方法
func (m Mobile) read() {
    fmt.Printf("Mobile %v is reading data\n", m.name)
    fmt.Println("reading...")
}

// 使用Mobile结构体实现USB接口的write方法
func (m Mobile) write() {
    fmt.Printf("Mobile %v is writing data\n", m.name)
    fmt.Println("writing...")
}

func main() {
    c := Computer{"Mac"}
    c.read()
    c.write()

    fmt.Println("================================")

    m := Mobile{"iPhone"}
    m.read()
    m.write()
}
    执行结果
    Computer Mac is reading data
reading...
Computer Mac is writing data
writing...
================================
Mobile iPhone is reading data
reading...
Mobile iPhone is writing data
writing...

    接口值类型接收者和指针类型接收者¶

    这个话题本质上和方法的值类型接收者和指针类型接收者是一致的，值接收者是一个拷贝，传递的是副本，而指针接收者，传递的是指针

    定义一个Pet接口

    type Pet interface {
    eat(string) string
}

    定义一个Dog结构体

    type Dog struct {
    name string
}

    实现Pet接口方法：值类型接收者

    func (d Dog) eat(name string) string {
    fmt.Printf("dog: %v is eating %v\n", d.name, name)
    fmt.Printf("dog: %p\n", &d)
    fmt.Println("dog is eating...")
    d.name = "黑黑"
    return d.name
}

    完整实例：

    package main

import "fmt"

// 定义一个Pet接口
type Pet interface {
    // 定义一个eat方法
    eat(string) string
}

// 定义一个Dog结构体
type Dog struct {
    name string
}

// 使用Dog结构体实现Pet接口的eat方法
func (d Dog) eat(name string) string {
    fmt.Printf("dog: %v is eating %v\n", d.name, name)
    fmt.Printf("dog: %p\n", &d)
    fmt.Println("dog is eating...")
    d.name = "黑黑"
    return d.name
}

func main() {
    dog := Dog{"旺财"}
    fmt.Printf("dog: %p\n", &dog)
    dog.eat("骨头")
    fmt.Printf("dog: %v\n", dog.name)

}
    执行结果
    dog: 0xc000014250
dog: 旺财 is eating 骨头
dog: 0xc000014260
dog is eating...
dog: 旺财

    定义一个Pet接口

    type Pet interface {
    eat(string) string
}

    定义一个Dog结构体

    type Dog struct {
    name string
}

    实现Pet接口方法：指针类型接收者

    func (d *Dog) eat(name string) string {
    fmt.Printf("dog: %v is eating %v\n", d.name, name)
    fmt.Printf("dog: %p\n", d)
    fmt.Println("dog is eating...")
    d.name = "黑黑"
    return d.name
}

    完整实例：

    package main

import "fmt"

// 定义一个Pet接口
type Pet interface {
    // 定义一个eat方法
    eat(string) string
}

// 定义一个Dog结构体
type Dog struct {
    name string
}

// 使用Dog结构体实现Pet接口的eat方法
func (d *Dog) eat(name string) string {
    fmt.Printf("dog: %v is eating %v\n", d.name, name)
    fmt.Printf("dog: %p\n", d)
    fmt.Println("dog is eating...")
    d.name = "黑黑"
    return d.name
}

func main() {
    dog := Dog{"旺财"}
    fmt.Printf("dog: %p\n", &dog)
    dog.eat("骨头")
    fmt.Printf("dog: %v\n", dog.name)

}
    执行结果
    dog: 0xc000014250
dog: 旺财 is eating 骨头
dog: 0xc000014250
dog is eating...
dog: 黑黑

    接口和类型的关系¶

    • 一个类型可以实现多个接口
    • 多个类型可以实现同一个接口

    一个类型实现多个接口¶

    一个类型实现多个接口，例如：有一个Player接口可以播放音乐，有一个Video接口可以播放视频，一个手机Mobile实现这两个接口，既可以播放音乐，又可以播放视频

    定义一个Player接口

    type Player interface {
    playMusic(string) string
}

    定义一个Video接口

    type Video interface {
    playVideo(string) string
}

    定义一个Mobile结构体

    type Mobile struct {
    name string
}

    实现Player和Video接口

    func (m Mobile) playMusic(name string) string {
    fmt.Printf("Mobile %v is playing music %v\n", m.name, name)
    fmt.Println("playing music...")
    return name
}

func (m Mobile) playVideo(name string) string {
    fmt.Printf("Mobile %v is playing video %v\n", m.name, name)
    fmt.Println("playing video...")
    return name
}

    完整实例：

    package main

import "fmt"

// 一个类型实现多个接口
type Player interface {
    playMusic(string) string
}

type Video interface {
    playVideo(string) string
}

// 定义一个Mobile结构体
type Mobile struct {
    name string
}

// 使用Mobile结构体实现两个接口
func (m Mobile) playMusic(name string) string {
    fmt.Printf("Mobile %v is playing music %v\n", m.name, name)
    fmt.Println("playing music...")
    return name
}

func (m Mobile) playVideo(name string) string {
    fmt.Printf("Mobile %v is playing video %v\n", m.name, name)
    fmt.Println("playing video...")
    return name
}


func main() {
    m := Mobile{"iPhone"}
    m.playMusic("青花瓷")
    m.playVideo("复仇者联盟")
}
    执行结果
    Mobile iPhone is playing music 青花瓷
playing music...
Mobile iPhone is playing video 复仇者联盟
playing video...

    多个类型实现一个接口¶

    比如，一个宠物接口Pet，猫类型Cat和狗类型Dog都可以实现该接口，都可以把猫和狗当宠物类型对待，这在其他语言中叫多态

    定义一个Pet接口

    type Pet interface {
    eat(string)
}

    定义一个Dog结构体

    type Dog struct {
    name string
}

    Dog实现Pet接口

    func (d Dog) eat(name string) {
    fmt.Printf("dog: %v is eating %v\n", d.name, name)
    fmt.Println("dog is eating...")
}

    定义一个Cat结构体

    type Cat struct {
    name string
}

    Cat实现Pet接口

    func (c Cat) eat(name string) {
    fmt.Printf("cat: %v is eating %v\n", c.name, name)
    fmt.Println("cat is eating...")
}

    具体实例：

    package main

import "fmt"

// 多个类型实现一个接口
type Pet interface {
    eat(string)
}

// 定义一个Dog结构体
type Dog struct {
    name string
}

// 使用Dog结构体实现Pet接口的eat方法
func (d Dog) eat(name string) {
    fmt.Printf("dog: %v is eating %v\n", d.name, name)
    fmt.Println("dog is eating...")
}

// 定义一个Cat结构体
type Cat struct {
    name string
}

// 使用Cat结构体实现Pet接口的eat方法
func (c Cat) eat(name string) {
    fmt.Printf("cat: %v is eating %v\n", c.name, name)
    fmt.Println("cat is eating...")
}

func main() {
    dog := Dog{"旺财"}
    dog.eat("骨头")

    cat := Cat{"多多"}
    cat.eat("猫条")
}
    执行结果
    dog: 旺财 is eating 骨头
dog is eating...
cat: 多多 is eating 猫条
cat is eating...

    接口嵌套¶

    接口可以通过嵌套，创建新的接口，例如：飞鱼，既可以飞，又可以游泳。我们创建一个Fly接口，创建一个Swim接口，飞鱼接口由这两个接口组成

    定义Fly接口

    type Fly interface {
    // fly方法
    fly()
}

    定义Swim接口

    type Swim interface {
    // swim方法
    swim()
}

    组合创建新接口FlyFish

    type FlyFish interface {
    Fly
    Swim
}

    定义Fish结构体

    type Fish struct {
    name string
}

    实现接口方法

    func (f Fish) fly() {
    fmt.Printf("fish: %v is flying\n", f.name)
}

func (f Fish) swim() {
    fmt.Printf("fish: %v is swimming\n", f.name)
}

    完整示例：

    package main

import "fmt"

// 接口嵌套
type Fly interface {
    fly()
}

type Swim interface {
    swim()
}

// 定义一个接口, 该接口中嵌套了两个接口
type FlyFish interface {
    Fly
    Swim
}

// 定义一个结构体
type Fish struct {
    name string
}

// 使用Fish结构体实现FlyFish接口的fly方法
func (f Fish) fly() {
    fmt.Printf("fish: %v is flying\n", f.name)
}

func (f Fish) swim() {
    fmt.Printf("fish: %v is swimming\n", f.name)
}


func main() {
    f := Fish{"多多鱼"}
    f.fly()
    f.swim()
}
    执行结果
    fish: 多多鱼 is flying
fish: 多多鱼 is swimming

    通过接口实现OCP设计原则¶

    面向对象的可复用设计的第一块基石，便是所谓的"开-闭"原则（Open-Closed Principle），虽然Golang不是面向对象语言，但是也可以模拟实现这个原则

    定义一个Pet接口

    type Pet interface {
    eat()
    sleep()
}

    该接口有吃和睡两个方法

    定义Dog结构体

    type Dog struct {
    name string
    age int
}

    Dog实现接口方法

    func (d Dog) eat() {
    fmt.Printf("dog: %v is eating\n", d.name)
}

func (d Dog) sleep() {
    fmt.Printf("dog: %v is sleeping\n", d.name)
}

    定义Cat结构体

    type Cat struct {
    name string
    age int
}

    Cat实现接口方法

    func (c Cat) eat() {
    fmt.Printf("cat: %v is eating\n", c.name)
}

func (c Cat) sleep() {
    fmt.Printf("cat: %v is sleeping\n", c.name)
}

    定义Person结构体

    type Person struct {
    name string
}

    为Person添加养宠物方法

    func (p Person) care(pet Pet) {
    p.eat()
    p.sleep()
}

    完整实例：

    package main

import "fmt"

// OCP开闭原则
type Pet interface {
    eat()
    sleep()
}

// 定义一个Dog结构体
type Dog struct {
    name string
    age  int
}

// Dog实现Pet接口的eat方法
func (d Dog) eat() {
    fmt.Printf("dog: %v is eating\n", d.name)
}

// Dog实现Pet接口的sleep方法
func (d Dog) sleep() {
    fmt.Printf("dog: %v is sleeping\n", d.name)
}

// 定义一个Cat结构体
type Cat struct {
    name string
    age  int
}

// Cat实现Pet接口的eat方法
func (c Cat) eat() {
    fmt.Printf("cat: %v is eating\n", c.name)
}

func (c Cat) sleep() {
    fmt.Printf("cat: %v is sleeping\n", c.name)
}

// 定义一个Person结构体
type Person struct {
    name string
}

// 为Person添加养宠物方法
func (p Person) care(pet Pet) {
    pet.eat()
    pet.sleep()
}

func main() {
    d := Dog{"旺财", 3}
    cat := Cat{"多多", 2}
    p := Person{"小明"}

    // 调用Person的care方法, 传入dog
    p.care(d)
    // 调用Person的care方法, 传入cat
    p.care(cat)
}
    执行结果
    dog: 旺财 is eating
dog: 旺财 is sleeping
cat: 多多 is eating
cat: 多多 is sleeping

    模拟OOP的属性和方法¶

    Golang没有面向对象的概念，也没有封装的概念，但是可以通过结构体struct和函数 绑定 来实现OOP的属性和方法等特性。

    例如，想要定义一个Person类，有name和age属性，有eat/sleep/work方法

    package main

import "fmt"

// OOP: Object Oriented Programming 面向对象编程
type Person struct {
    // 定义name和age属性
    name string
    age  int
}

// 为Person结构体添加方法
func (p Person) eat() {
    fmt.Printf("%v is eating\n", p.name)
}

func (p Person) sleep() {
    fmt.Printf("%v is sleeping\n", p.name)
}

func (p Person) work() {
    fmt.Printf("%v is working\n", p.name)
}

func main() {
    p := Person{"小明", 18}

    // 访问Person的属性
    fmt.Println(p.name, p.age)

    // 调用Person的eat方法
    p.eat()
    // 调用Person的sleep方法
    p.sleep()
    // 调用Person的work方法
    p.work()
}
    执行结果
    小明 18
小明 is eating
小明 is sleeping
小明 is working
